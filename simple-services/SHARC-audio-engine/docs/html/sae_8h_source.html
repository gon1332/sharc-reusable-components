122/*!****************************************************************
123 * @brief SHARC Audio Engine initialization
124 *
125 * This function allocates and initializes the SHARC Audio Engine
126 * on a core.
127 *
128 * This function must be called by each participating core. Each
129 * core must be assigned a sequential core index. One core must be
130 * delegated as the master core.
131 *
132 * The master core is responsible for global SoC wide initialization.
133 * The master core must run this function to completion before the
134 * initialization of any other core.
135 *
136 * This function is not thread safe.
137 *
138 * @param [out] context A pointer to an SAE context pointer
139 * @param [in] saeIdx Index assigned by the application to this core
140 * @param [in] saeMaster Set to true only on the SAE master core
141 *
142 * @return Returns SAE_RESULT_OK if successful, otherwise
143 * an error.
144 ******************************************************************/
145SAE_RESULT sae_initialize(SAE_CONTEXT **context, SAE_CORE_IDX saeIdx,
146 bool saeMaster);
